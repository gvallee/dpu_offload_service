697 #define SMART_BUFF_GET(__sys, __sz) \
698  ({ \
699  size_t _i; \
700  smart_bucket_t *_target_bucket = NULL; \
701  smart_chunk_t *_target_chunk = NULL; \
702  for (_i = 0; _i < (__sys)->num_buckets; _i++) \
703  { \
704  smart_bucket_t *__b = DYN_ARRAY_GET_ELT(&((__sys)->buckets), _i, smart_bucket_t); \
705  assert(__b); \
706  if ((__b)->min_size <= (__sz) && (__sz) <= (__b)->max_size) \
707  { \
708  _target_bucket = __b; \
709  break; \
710  } \
711  } \
712  if (_target_bucket == NULL || _target_bucket->max_size == UINT64_MAX) \
713  { \
714  /* Need to create a new bucket, the size is bigger than anything we handle right now */ \
715  /* We take the larger size from the next size aligned on a page or twice the maximum size */ \
716  /* of the previous bucket */ \
717  size_t *__last_bucket_sz = DYN_ARRAY_GET_ELT(&((__sys)->bucket_sizes), \
718  (__sys)->num_buckets - 1, \
719  size_t); \
720  uint64_t _new_bucket_max_size = (((__sz) / 4096) + 1) * 4096; \
721  if (_new_bucket_max_size < *__last_bucket_sz * 2) \
722  _new_bucket_max_size = *__last_bucket_sz * 2; \
723  size_t *__new_bucket_sz; \
724  smart_bucket_t *__new_bucket; \
725  mem_chunk_t *_mem_chunk_desc; \
726  __new_bucket_sz = DYN_ARRAY_GET_ELT(&((__sys)->bucket_sizes), (__sys)->num_buckets, size_t); \
727  assert(__new_bucket_sz); \
728  *__new_bucket_sz = _new_bucket_max_size; \
729  __new_bucket = DYN_ARRAY_GET_ELT(&((__sys)->buckets), (__sys)->num_buckets, smart_bucket_t); \
730  assert(__new_bucket); \
731  SIMPLE_LIST_INIT(&(__new_bucket->pool)); \
732  __new_bucket->min_size = (uint64_t)(*__last_bucket_sz + 1); \
733  __new_bucket->max_size = _new_bucket_max_size; \
734  _mem_chunk_desc = DYN_ARRAY_GET_ELT(&((__sys)->base_mem_chunks), \
735  (__sys)->num_base_mem_chunks, \
736  mem_chunk_t); \
737  assert(_mem_chunk_desc); \
738  /* Calculate the total number of memory for the bucket */ \
739  size_t _bucket_mem_size = 2 * _new_bucket_max_size; \
740  _mem_chunk_desc->ptr = calloc(_bucket_mem_size, sizeof(char)); \
741  (__sys)->num_base_mem_chunks++; \
742  /* Add the smart chunks */ \
743  size_t _j, __n = 0; \
744  for (_j = 0; _j <= (_bucket_mem_size / __new_bucket->max_size); _j++) \
745  { \
746  void *_mem_ptr = (void *)((ptrdiff_t)_mem_chunk_desc->ptr + _j * __new_bucket->max_size); \
747  smart_chunk_t *_smart_chunk; \
748  DYN_LIST_GET((__sys)->smart_chunk_desc_pool, smart_chunk_t, super, _smart_chunk); \
749  assert(_smart_chunk); \
750  RESET_SMART_CHUNK(_smart_chunk); \
751  _smart_chunk->base = _mem_ptr; \
752  _smart_chunk->size = __new_bucket->max_size; \
753  SIMPLE_LIST_PREPEND(&(__new_bucket->pool), &(_smart_chunk->super)); \
754  __n++; \
755  } \
756  __new_bucket->initial_size = __n; \
757  (__sys)->num_buckets++; \
758  _target_bucket = __new_bucket; \
759  } \
760  assert(_target_bucket); \
761  _target_chunk = SIMPLE_LIST_EXTRACT_HEAD(&(_target_bucket->pool), smart_chunk_t, super); \
762  _target_chunk->in_use = true; \
763  _target_chunk; \
764  })
765 
766 #define SMART_BUFF_RETURN(__sys, __sz, __sc) ({ \
767  size_t _sb_return_i; \
768  smart_bucket_t *_target_bucket = NULL; \
769  assert(__sc->in_use == true); \
770  /* Find the associated bucket */ \
771  for (_sb_return_i = 0; _sb_return_i < (__sys)->num_buckets; _sb_return_i++) \
772  { \
773  smart_bucket_t *__b = DYN_ARRAY_GET_ELT(&((__sys)->buckets), _sb_return_i, smart_bucket_t); \
774  assert(__b); \
775  if ((__b)->min_size <= __sz && __sz <= (__b)->max_size) \
776  { \
777  _target_bucket = __b; \
778  break; \
779  } \
780  } \
781  assert(_target_bucket); \
782  __sc->in_use = false; \
783  SIMPLE_LIST_PREPEND(&(_target_bucket->pool), &(__sc->super)); \
784  /* Check if it makes sense to try to recycle smart buffers */ \
785  size_t _recycle_threshold = _target_bucket->initial_size + _target_bucket->initial_size / 2; \
786  if (SIMPLE_LIST_LENGTH(&(_target_bucket->pool)) > _recycle_threshold) \
787  SMART_BUFFS_RECYCLE_SMART_CHUNK(__sc); \
788  /* Make sure the pointer cannot be used anymore */ \
789  __sc = NULL; \
790 })

823  assert(__mc); \

827  } while (0)

919  // type, the update this pointer to be able to return it when the
920  // smart list is freed.

934  assert(__smart_list); \

936  /* Get a new buffer from the smart buffer system, no allocation */ \
937  size_t _chunk_size = (__smart_list)->num_elts_alloc * (__smart_list)->elt_size; \
938  smart_chunk_t *_new_smart_chunk = (smart_chunk_t *)SMART_BUFF_GET((__smart_buf_sys), _chunk_size); \
939  if (_new_smart_chunk != NULL) \
940  { \
941  size_t _i; \

953  for (_i = 0; _i < (__smart_list)->num_elts_alloc; _i++) \
954  { \

956  _ptr = &(_e[_i]); \
957  } \
958  assert(SIMPLE_LIST_LENGTH(&((__smart_list)->list)) == \
959  (_initial_list_size + (__smart_list)->num_elts_alloc)); \
960  /* All the new elements are added to the list, update the number of elements in the list*/ \
961  (__smart_list)->num_elts += (__smart_list)->num_elts_alloc; \
962  } \
963  else \
964  { \
965  fprintf(stderr, "[ERROR] unable to grow smart list, unable to allocate buffer\n"); \
966  } \

1019  *_smart_chunk_ptr = NULL; \

1026 #define SMART_LIST_GET(_smart_buf_sys, _smart_list, _type, _elt, _item) \
