21 /*
22  * UCP tag structure:
23  *
24  * 01 | 01234567 01234567 | 234 | 567 | 01234567 01234567 01234567 | 01234567 01234567
25  * | | | | |
26  * RESERV(2) | message tag (16) | SCOPE(3) | SCOPE_ID(3) | source rank (24) | team id (16)
27  */

138  } \
139  else \
140  { \
141  /* We are dealing with a sub-communicator. In that context, _n_local_ranks is equal to the */ \

143  if (_n_local_ranks == __gp_cache->n_local_ranks_populated) \

146  broadcast_group_cache((_engine), (_gp_id)); \
147  } \
148  } \
149  \
150  /* We do not have enough information about the group to do anything optimized, we just broadcast */ \
151  /* with what we got. */ \
152  if (__gp_cache->n_local_ranks < 0) \

154  DBG("We do not know how many ranks to locally expect for the group, broadcast by default"); \

227 static bool event_posted(dpu_offload_event_t *ev)
228 {
229  if (ev->ctx.hdr_completed == false && ev->hdr_request != NULL && EVENT_HDR_PAYLOAD_SIZE(ev) == 0)
230  {
231  // The send for the header is posted (and not completed) and no payload needs to be sent
232  return true;
233  }
234 
235  if (ev->ctx.hdr_completed == true &&
236  ev->hdr_request == NULL &&
237  EVENT_HDR_PAYLOAD_SIZE(ev) > 0 &&
238  ev->ctx.payload_completed == false &&
239  ev->payload_request != NULL)
240  {
241  // The send for the header completed and the send for the payload was posted and not yet completed
242  return true;
243  }
244 
245  return false;
246 }

249 static void progress_econtext_sends(execution_context_t *ctx)
250 {
251  dpu_offload_event_t *ev, *next_ev;
252 #if USE_AM_IMPLEM
253  if (ctx->scope_id == CONTEXT_SELF)
254  return;
255 #endif
256  ucs_list_for_each_safe(ev, next_ev, &(ctx->ongoing_events), item)
257  {
258  if (ev->is_ongoing_event == false)
259  {
260  ERR_MSG("Ev %p type %ld is not on ongoing list", ev, EVENT_HDR_TYPE(ev));
261  assert(0);
262  }
263  assert(ev->is_ongoing_event == true);
264  if (EVENT_HDR_TYPE(ev) == META_EVENT_TYPE)
265  {
266  assert(ev->sub_events_initialized);
267  // if the event is meta-event, we need to check the sub-events
268  dpu_offload_event_t *subev, *next_subev;
269  ucs_list_for_each_safe(subev, next_subev, (&(ev->sub_events)), item)
270  {
271  PROGRESS_EVENT_SEND(subev);
272  }
273 
274  // Finally check if the meta-event is now completed
275  if (event_completed(ev))
276  {
277  assert(ev->is_ongoing_event);
278  ucs_list_del(&(ev->item));
279  ev->is_ongoing_event = false;
280  if (ev->was_posted)
281  {
282  ev->event_system->posted_sends--;
283  ev->was_posted = false;
284  }
285  event_return(&ev);
286  }
287  }
288  else
289  {
290  PROGRESS_EVENT_SEND(ev);
291  }
292  }

307 static int handle_notif_msg(execution_context_t *econtext, am_header_t *hdr, size_t header_length, void *data, size_t length)
308 {
309  assert(econtext);
310  assert(hdr);
311  assert(econtext->event_channels);
312  if (hdr->payload_size > 0 && data == NULL)
313  {
314  ERR_MSG("the payload is %" PRIu64 " but the buffer is NULL", hdr->payload_size);
315  return UCS_ERR_NO_MESSAGE;
316  }
317  SYS_EVENT_LOCK(econtext->event_channels);
318  notification_callback_entry_t *entry = DYN_ARRAY_GET_ELT(&(econtext->event_channels->notification_callbacks), hdr->type, notification_callback_entry_t);
319  DBG("Notification of type %" PRIu64 " received from %" PRIu64 " (econtext: %p), dispatching...", hdr->type, hdr->id, econtext);
320  if (entry->set == false)
321  {
322  pending_notification_t *pending_notif;
323  DBG("callback not available for %" PRIu64 " on event system %p (econtext: %p)",
324  hdr->type, econtext->event_channels, econtext);
325  DYN_LIST_GET(econtext->event_channels->free_pending_notifications, pending_notification_t, item, pending_notif);
326  CHECK_ERR_RETURN((pending_notif == NULL), UCS_ERR_NO_MESSAGE, "unable to get pending notification object");
327  RESET_PENDING_NOTIF(pending_notif);
328  pending_notif->type = hdr->type;
329  pending_notif->src_id = hdr->id;
330  pending_notif->data_size = length;
331  pending_notif->header_size = header_length;
332  pending_notif->econtext = econtext;
333  if (pending_notif->data_size > 0)
334  {
335  pending_notif->data = DPU_OFFLOAD_MALLOC(pending_notif->data_size);
336  CHECK_ERR_RETURN((pending_notif->data == NULL), DO_ERROR, "unable to allocate pending notification's data");
337  memcpy(pending_notif->data, data, pending_notif->data_size);
338  }
339  else
340  {
341  pending_notif->data = NULL;
342  }
343  if (pending_notif->header_size > 0)
344  {
345  pending_notif->header = DPU_OFFLOAD_MALLOC(pending_notif->header_size);
346  CHECK_ERR_RETURN((pending_notif->header == NULL), DO_ERROR, "unable to allocate pending notification's header");
347  memcpy(pending_notif->header, hdr, pending_notif->header_size);
348  }
349  else
350  {
351  pending_notif->header = NULL;
352  }
353  ucs_list_add_tail(&(econtext->event_channels->pending_notifications), &(pending_notif->item));
354  SYS_EVENT_UNLOCK(econtext->event_channels);
355  return UCS_OK;
356  }
357 
358  notification_cb cb = entry->cb;
359  CHECK_ERR_GOTO((cb == NULL), error_out, "Callback is undefined");
360  struct dpu_offload_ev_sys *ev_sys = EV_SYS(econtext);
361  // Callbacks are responsible for handling any necessary locking
362  // and can call any event API so we unlock before invoking it.
363  SYS_EVENT_UNLOCK(econtext->event_channels);
364  // We unlock the execution context before invoking the callback to limit
365  // the constraints on what can be done in the callback.
366  cb(ev_sys, econtext, hdr, header_length, data, length);
367  return UCS_OK;
368 error_out:
369  SYS_EVENT_UNLOCK(econtext->event_channels);
370  return UCS_ERR_NO_MESSAGE;
371 }

382 static void notif_payload_recv_handler(void *request, ucs_status_t status, const ucp_tag_recv_info_t *tag_info, void *user_data)
383 {
384  int rc;
385  hdr_notif_req_t *ctx;
386  if (status == UCS_ERR_CANCELED)
387  {
388  // Callback was invoked during finalization when the notification recv is canceled. Do nothing.
389  return;
390  }
391  assert(status == UCS_OK);
392  ctx = (hdr_notif_req_t *)user_data;
393  assert(ctx);
394  assert(ctx->econtext);
395  DBG("Notification payload received, ctx=%p econtext=%p type=%ld", ctx, ctx->econtext, ctx->hdr.type);
396  assert(ctx->hdr.payload_size == tag_info->length);
397  ctx->payload_ctx.complete = true;
398 
399  // Invoke the associated callback
400  rc = handle_notif_msg(ctx->econtext, &(ctx->hdr), sizeof(am_header_t), ctx->payload_ctx.buffer, ctx->hdr.payload_size);
401  if (rc != UCS_OK)
402  {
403  ERR_MSG("handle_notif_msg() failed");
404  assert(0); // fail when in debug mode
405  return;
406  }
407 
408  if (ctx->req != NULL)
409  {
410  assert(ctx->complete == true);
411  ucp_request_free(ctx->req);
412  ctx->req = NULL;
413  }
414  if (ctx->payload_ctx.req != NULL)
415  {
416  assert(ctx->payload_ctx.complete == true);
417  ucp_request_free(ctx->payload_ctx.req);
418  ctx->payload_ctx.req = NULL;
419  }
420  if (ctx->payload_ctx.buffer != NULL)
421  {
422  if (ctx->payload_ctx.pool.mem_pool != NULL)
423  {
424  if (ctx->payload_ctx.pool.return_buf != NULL)
425  {
426  // Call the return function specified by the caller library.
427  assert(ctx->payload_ctx.pool.return_buf);
428  ctx->payload_ctx.pool.return_buf(ctx->payload_ctx.pool.mem_pool,
429  ctx->payload_ctx.buffer);
430  }
431  ctx->payload_ctx.pool.mem_pool = NULL;
432  }
433  else
434  {
435 #if BUDDY_BUFFER_SYS_ENABLE
436  // Return the buffer to the smart buffer system
437  assert(ctx->payload_ctx.smart_buf);
438  SMART_BUFF_RETURN(&(ctx->econtext->engine->smart_buffer_sys),
439  ctx->hdr.payload_size,
440  ctx->payload_ctx.smart_buf);
441  ctx->payload_ctx.smart_buf = NULL;
442 #else
443  free(ctx->payload_ctx.buffer);
444  ctx->payload_ctx.buffer = NULL;
445 #endif
446  }
447  ctx->payload_ctx.buffer = NULL;
448  }
449  ctx->payload_ctx.complete = true;
450 }

460 static int post_recv_for_notif_payload(hdr_notif_req_t *ctx, execution_context_t *econtext, uint64_t peer_id)
461 {
462  int rc = EVENT_INPROGRESS;
463  assert(ctx);
464  assert(econtext);
465  DBG("Notification header received, payload size = %ld, type = %ld, econtext = %p-%p, ctx = %p",
466  ctx->hdr.payload_size,
467  ctx->hdr.type,
468  ctx->econtext,
469  econtext,
470  ctx);
471  assert(econtext == ctx->econtext);
472  assert(ctx->complete == true); // The header should be completed
473 
474  if (ctx->hdr.payload_size > 0 && ctx->payload_ctx.req != NULL)
475  {
476  DBG("We got a new header but still waiting for a previous notification payload");
477  return EVENT_INPROGRESS;
478  }
479 
480  if (ctx->hdr.payload_size > 0)
481  {
482  ucp_tag_t payload_ucp_tag, payload_ucp_tag_mask;
483  notification_callback_entry_t *entry;
484  DBG("Posting recv for notif payload of size %ld for peer %ld (client_id: %" PRIu64 ", server_id: %" PRIu64 ")",
485  ctx->hdr.payload_size, peer_id, ctx->client_id, ctx->server_id);
486  ucp_worker_h worker;
487 
488  // If the notification type is already registered and is associated to a memory pool, we use a buffer from than pool
489  entry = DYN_ARRAY_GET_ELT(&(econtext->event_channels->notification_callbacks), ctx->hdr.type, notification_callback_entry_t);
490  assert(entry);
491  if (entry->info.mem_pool)
492  {
493  void *buf_from_pool = get_notif_buf(econtext->event_channels, ctx->hdr.type);
494  assert(buf_from_pool);
495  assert(ctx->payload_ctx.pool.mem_pool == NULL);
496  ctx->payload_ctx.buffer = buf_from_pool;
497  COPY_NOTIF_INFO(&(entry->info), &(ctx->payload_ctx.pool));
498  }
499  else
500  {
501  // Get a buffer from the smart buffer system to avoid allocating memory
502  assert(ctx->payload_ctx.smart_buf == NULL);
503 #if BUDDY_BUFFER_SYS_ENABLE
504  ctx->payload_ctx.smart_buf = SMART_BUFF_GET(&(econtext->engine->smart_buffer_sys), ctx->hdr.payload_size);
505  assert(ctx->payload_ctx.smart_buf);
506  ctx->payload_ctx.buffer = ctx->payload_ctx.smart_buf->base;
507 #else
508  ctx->payload_ctx.buffer = malloc(ctx->hdr.payload_size);
509 #endif
510  }
511  assert(ctx->payload_ctx.buffer);
512  worker = GET_WORKER(econtext);
513  assert(worker);
514  // Post the receive for the payload
515  ctx->payload_ctx.complete = false;
516  memset(&ctx->payload_ctx.recv_params, 0, sizeof(ctx->payload_ctx.recv_params));
517  ctx->payload_ctx.recv_params.op_attr_mask = UCP_OP_ATTR_FIELD_CALLBACK |
518  UCP_OP_ATTR_FIELD_DATATYPE |
519  UCP_OP_ATTR_FIELD_USER_DATA;
520  ctx->payload_ctx.recv_params.datatype = ucp_dt_make_contig(1);
521  ctx->payload_ctx.recv_params.user_data = ctx;
522  ctx->payload_ctx.recv_params.cb.recv = notif_payload_recv_handler;
523  MAKE_RECV_TAG(payload_ucp_tag, payload_ucp_tag_mask, AM_EVENT_MSG_ID, ctx->client_id, ctx->server_id, econtext->scope_id, 0);
524  DBG("Tag: %d; scope_id: %u", AM_EVENT_MSG_ID, econtext->scope_id);
525  ctx->payload_ctx.req = ucp_tag_recv_nbx(worker,
526  ctx->payload_ctx.buffer,
527  ctx->hdr.payload_size,
528  payload_ucp_tag,
529  payload_ucp_tag_mask,
530  &(ctx->payload_ctx.recv_params));
531  if (ctx->payload_ctx.req == NULL)
532  {
533  int rc;
534  // Recv completed immediately, the callback is not invoked
535  DBG("Recv of the payload completed right away");
536  ctx->payload_ctx.complete = true;
537  rc = handle_notif_msg(ctx->econtext, &(ctx->hdr), sizeof(am_header_t), ctx->payload_ctx.buffer, ctx->hdr.payload_size);
538  if (rc != UCS_OK)
539  {
540  ERR_MSG("handle_notif_msg() failed");
541  return -1;
542  }
543 
544  if (ctx->payload_ctx.buffer != NULL)
545  {
546  if (ctx->payload_ctx.pool.mem_pool == NULL)
547  {
548 #if BUDDY_BUFFER_SYS_ENABLE
549  // Return the smart chunk to the smart buffer system.
550  assert(ctx->payload_ctx.smart_buf);
551  SMART_BUFF_RETURN(&(econtext->engine->smart_buffer_sys),
552  ctx->hdr.payload_size,
553  ctx->payload_ctx.smart_buf);
554  ctx->payload_ctx.smart_buf = NULL;
555  assert(ctx->payload_ctx.smart_buf == NULL);
556 #else
557  free(ctx->payload_ctx.buffer);
558  ctx->payload_ctx.buffer = NULL;
559 #endif
560  }
561  if (ctx->payload_ctx.pool.mem_pool != NULL)
562  {
563  if (ctx->payload_ctx.pool.return_buf != NULL)
564  {
565  // The calling library as its own pool of objects to handle notifications.
566  // Call the return function.
567  ctx->payload_ctx.pool.return_buf(ctx->payload_ctx.pool.mem_pool, ctx->payload_ctx.buffer);
568  }
569  ctx->payload_ctx.pool.mem_pool = NULL;
570  }
571  ctx->payload_ctx.buffer = NULL;
572  }
573 #if !NDEBUG
574  else
575  {
576  assert(ctx->hdr.payload_size == 0);
577  }
578 #endif
579  assert(ctx->payload_ctx.smart_buf == NULL);
580 
581  if (ctx->req != NULL)
582  {
583  assert(ctx->complete == true);
584  ucp_request_free(ctx->req);
585  ctx->req = NULL;
586  }
587  if (ctx->payload_ctx.req != NULL)
588  {
589  assert(ctx->payload_ctx.complete == true);
590  ucp_request_free(ctx->payload_ctx.req);
591  ctx->payload_ctx.req = NULL;
592  }
593  rc = EVENT_DONE;
594  assert(ctx->payload_ctx.smart_buf == NULL);
595  }
596  else
597  rc = EVENT_INPROGRESS;
598 
599  if (UCS_PTR_IS_ERR(ctx->payload_ctx.req))
600  {
601  ucs_status_t recv_status = UCS_PTR_STATUS(ctx->payload_ctx.req);
602  ERR_MSG("ucp_tag_recv_nbx() failed: %s", ucs_status_string(recv_status));
603  return -1;
604  }
605  }
606  else
607  {
608  // No payload to be received
609  ctx->payload_ctx.complete = true;
610  int rc = handle_notif_msg(econtext, &(ctx->hdr), sizeof(am_header_t), NULL, 0);
611  if (rc != UCS_OK)
612  {
613  ERR_MSG("handle_notif_msg() failed\n");
614  return -1;
615  }
616 
617  // Payload is freed when the event is returned
618  if (ctx->req != NULL)
619  {
620  ucp_request_free(ctx->req);
621  ctx->req = NULL;
622  }
623  if (ctx->payload_ctx.req != NULL)
624  {
625  ucp_request_free(ctx->payload_ctx.req);
626  ctx->payload_ctx.req = NULL;
627  }
628 
629  rc = EVENT_DONE;
630  }
631  return rc;
632 }

645 static int post_new_notif_recv(ucp_worker_h worker, hdr_notif_req_t *ctx, execution_context_t *econtext, ucp_tag_t hdr_ucp_tag, ucp_tag_t hdr_ucp_tag_mask, ucp_request_param_t *hdr_recv_param)
646 {
647  int rc = EVENT_INPROGRESS;
648 #if !NDEBUG
649  if (econtext->engine->on_dpu && econtext->scope_id == SCOPE_INTER_SERVICE_PROCS)
650  {
651  if (ctx->client_id >= econtext->engine->num_service_procs)
652  {
653  ERR_MSG("requested client ID is invalid: %" PRIu64, ctx->client_id);
654  }
655  assert(ctx->client_id < econtext->engine->num_service_procs);
656  }
657 #endif
658 
659  // If the execution context is in the process of finalizing, do not post a new receive.
660  // Only valid when the execution context is a client or a server.
661  if ((econtext->type == CONTEXT_SERVER || econtext->type == CONTEXT_CLIENT) && EXECUTION_CONTEXT_DONE(econtext) == true)
662  return EVENT_INPROGRESS;
663 
664  // Post a new receive only if we are not already in the middle of receiving a notification
665  if (ctx->complete == true && ctx->payload_ctx.complete == true)
666  {
667  // Post the receive for the header
668  ctx->complete = false;
669  ctx->econtext = (struct execution_context *)econtext;
670  DBG("-------------> Posting recv for notif header (econtext: %p, scope_id: %d, worker: %p, client_id: %" PRIu64 ", server_id: %" PRIu64 ", size: %ld)",
671  econtext, econtext->scope_id, worker, ctx->client_id, ctx->server_id, sizeof(am_header_t));
672  struct ucx_context *req = ucp_tag_recv_nbx(worker, &(ctx->hdr), sizeof(am_header_t), hdr_ucp_tag, hdr_ucp_tag_mask, hdr_recv_param);
673  if (req == NULL)
674  {
675  assert(ctx->client_id == ctx->hdr.client_id);
676  assert(ctx->server_id == ctx->hdr.server_id);
677 
678  // Receive completed immediately, callback is not called
679  DBG("Recv of notification header completed right away, notif type: %ld", ctx->hdr.type);
680  ctx->complete = true;
681  rc = post_recv_for_notif_payload(ctx, (execution_context_t *)ctx->econtext, ctx->hdr.id);
682  }
683  else
684  {
685  ctx->req = req;
686  rc = EVENT_INPROGRESS;
687  }
688  }
689  return rc;
690 }
